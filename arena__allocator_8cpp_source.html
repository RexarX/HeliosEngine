82auto ArenaAllocator::TryReserve(Block& block, size_t size,
83 size_t alignment) noexcept -> Reservation {
84 size_t current_offset = block.offset.load(std::memory_order_acquire);
85 for (;;) {
86 auto* const begin = static_cast<std::byte*>(block.buffer) + current_offset;
87 const size_t padding = CalculatePadding(begin, alignment);
88 const size_t aligned_offset = SaturatingAdd(current_offset, padding);
89 const size_t end_offset = SaturatingAdd(aligned_offset, size);
90
91 if (end_offset > block.capacity) {
92 return {};
93 }
94
95 if (block.offset.compare_exchange_weak(current_offset, end_offset,
96 std::memory_order_release,
97 std::memory_order_acquire)) {
98 return {
99 .ptr = static_cast<std::byte*>(block.buffer) + aligned_offset,
100 .size = size,
101 .padding = padding,
102 };
103 }

107bool ArenaAllocator::EnsureCapacity(size_t min_capacity) noexcept {
108 Block* observed_head = head_.load(std::memory_order_acquire);
109 const size_t current_capacity =
110 observed_head != nullptr ? observed_head->capacity : initial_capacity_;
111 const size_t desired_capacity =
112 growth_.NextCapacity(current_capacity, min_capacity);
113
114 if (desired_capacity < min_capacity || desired_capacity == 0) {
115 return false;
116 }
117
118 auto expected = GrowState::kIdle;
119 if (!grow_state_.compare_exchange_strong(expected, GrowState::kGrowing,
120 std::memory_order_acq_rel,
121 std::memory_order_acquire)) {
122 grow_state_.wait(GrowState::kGrowing, std::memory_order_acquire);
123 Block* const new_head = head_.load(std::memory_order_acquire);
124 return new_head != nullptr && new_head->capacity >= min_capacity;
125 }
126
127 Block* new_block = CreateBlock(desired_capacity);
128 if (new_block == nullptr) {
129 grow_state_.store(GrowState::kIdle, std::memory_order_release);
130 grow_state_.notify_all();
131 return false;
132 }
133
134 PublishBlock(new_block);
135 grow_state_.store(GrowState::kIdle, std::memory_order_release);
136 grow_state_.notify_all();
137 return true;
138}

152void* ArenaAllocator::do_allocate(size_t bytes, size_t alignment) {
153 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::ArenaAllocator::do_allocate");
154 HELIOS_MEMORY_PROFILE_ZONE_VALUE(bytes);
155
156 if (bytes == 0) [[unlikely]] {
157 return nullptr;
158 }
159
160 HELIOS_VERIFY(IsPowerOfTwo(alignment),
161 "alignment must be a power of two, got {}!", alignment);
162
163 const size_t effective_alignment = std::max(alignment, kMinAlignment);
164 Block* head = head_.load(std::memory_order_acquire);
165
166 Reservation reservation{};
167 if (head != nullptr) {
168 reservation = TryReserve(*head, bytes, effective_alignment);
169 }
170
171 if (reservation.ptr == nullptr) {
172 const size_t min_capacity = SaturatingAdd(bytes, effective_alignment);
173 HELIOS_VERIFY(EnsureCapacity(min_capacity),
174 "Unable to grow arena block chain!");
175
176 head = head_.load(std::memory_order_acquire);
177 HELIOS_ASSERT(head != nullptr, "Head block must exist after grow!");
178 reservation = TryReserve(*head, bytes, effective_alignment);
179 HELIOS_VERIFY(reservation.ptr != nullptr,
180 "Reservation failed after successful arena growth!");
181 }
182
183 allocation_count_.fetch_add(1, std::memory_order_relaxed);
184 total_allocations_.fetch_add(1, std::memory_order_relaxed);
185 alignment_waste_.fetch_add(reservation.padding, std::memory_order_relaxed);
186 const size_t total =
187 total_allocated_.fetch_add(bytes, std::memory_order_relaxed) + bytes;
188 AccumulatePeak(peak_usage_, total);
189
190 return reservation.ptr;
191}
