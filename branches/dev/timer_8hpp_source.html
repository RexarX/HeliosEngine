13/**
14 * @brief High–resolution timer with configurable clock and rich elapsed API.
15 * @details The timer measures elapsed time from the last reset point.
16 *
17 * - The clock type is configurable via the template parameter @p Clock.
18 * - Users can query elapsed time as:
19 * - A raw `Clock::duration`.
20 * - Any `std::chrono::duration` (via `ElapsedDuration`).
21 * - Arbitrary arithmetic type with specified time unit (via `Elapsed`).
22 * - Convenience helpers (seconds, milliseconds, microseconds, nanoseconds).
23 *
24 * @note The default clock is `std::chrono::steady_clock`.
25 * @tparam Clock Clock type used to measure time. Must satisfy
26 * `ClockTrait`.
27 */
28template <ClockTrait Clock = std::chrono::steady_clock>
29class Timer {
30public:

32 using TimePoint = typename Clock::time_point;
33 using Duration = typename Clock::duration;
