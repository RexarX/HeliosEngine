175/**
176 * @brief Concept for systems that declare all data access via parameter types.
177 * @details A system must be destructible, move or copy constructible, not
178 * polymorphic, an object type, and its `operator()` parameters must all have
179 * `SystemParamTraits` specialisations.
180 */
181template <typename T>
182concept SystemTrait =
183 std::destructible<T> &&
184 (std::move_constructible<T> || std::copy_constructible<T>) &&
185 !std::is_polymorphic_v<std::remove_cvref_t<T>> &&
186 std::is_object_v<std::remove_cvref_t<T>> &&
187 details::HasSystemParamCallOperator<T>;
188
189/**
190 * @brief Concept for lambda closure types that satisfy `SystemTrait`.
191 * @details Lambdas are detected via `utils::IsLambda<T>()`. They should always
192 * be registered with an explicit human-readable name.
193 */
194template <typename T>
195concept LambdaSystemTrait =
196 SystemTrait<T> && utils::LambdaTrait<std::remove_cvref_t<T>>;
