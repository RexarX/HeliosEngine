131constexpr auto PluginGroup::Configure(this auto&& self, T&& plugin)
132 -> decltype(std::forward<decltype(self)>(self)) {
133 using PluginType = std::remove_cvref_t<T>;
134 auto& group = static_cast<PluginGroup&>(self);
135
136 auto* storage = group.plugins_.template TryGet<PluginType>();
137 HELIOS_ASSERT(storage != nullptr, "Plugin '{}' is not part of this group!",
138 PluginNameOf<PluginType>());
139
140 if (storage == nullptr) [[unlikely]] {
141 return std::forward<decltype(self)>(self);
142 }
143
144 HELIOS_ASSERT(!storage->disabled, "Cannot configure disabled plugin '{}'!",
145 PluginNameOf<PluginType>());
146 if (storage->disabled) [[unlikely]] {
147 return std::forward<decltype(self)>(self);
148 }
149
150 *storage = PluginStorage::From<PluginType>(std::forward<T>(plugin));
151
152 return std::forward<decltype(self)>(self);
