38 * @details Requires nested types `rep`, `period`, `duration` and `time_point`,

91/**
92 * @brief Concept to allow argument conversion with polymorphic relationships.
93 * @details Allows regular convertibility or base/derived relationship for
94 * polymorphic types. This is useful for type-erased containers and delegates to
95 * provide flexible argument handling.
96 *
97 * The concept is satisfied when:
98 * - From is convertible to To (standard conversion), OR
99 * - Both From and To are polymorphic types and one derives from the other
100 *
101 * @tparam From Source type.
102 * @tparam To Target type.
103 */
104template <typename From, typename To>
105concept PolymorphicConvertible =
106 std::convertible_to<From, To> ||
107 (std::is_polymorphic_v<std::remove_reference_t<From>> &&
108 std::is_polymorphic_v<std::remove_reference_t<To>> &&
109 (std::derived_from<std::remove_cvref_t<From>, std::remove_cvref_t<To>> ||
110 std::derived_from<std::remove_cvref_t<To>, std::remove_cvref_t<From>>));
