29[[nodiscard]] void* RawAlloc(size_t size) noexcept {
30#ifdef HELIOS_GLOBAL_ALLOC_USE_MIMALLOC_BACKEND
31 return mi_malloc(size);
32#elif defined(_MSC_VER) && defined(_DEBUG)
33 // MSVC debug builds route new/delete through the debug heap. Using malloc
34 // here mixes allocators and can deadlock during static initialization.
35 return _malloc_dbg(size, _NORMAL_BLOCK, nullptr, 0);
36#else
37 return std::malloc(size);
38#endif
39}

41[[nodiscard]] void* RawAlignedAlloc(size_t size,
