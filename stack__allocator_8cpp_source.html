18void AccumulatePeak(std::atomic<size_t>& peak, size_t candidate) noexcept {
19 size_t observed = peak.load(std::memory_order_relaxed);
20 while (candidate > observed) {
21 if (peak.compare_exchange_weak(observed, candidate,
22 std::memory_order_relaxed,

33void StackAllocator::RewindToMarker(Marker marker) noexcept {
34 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::StackAllocator::RewindToMarker");
35 HELIOS_MEMORY_PROFILE_ZONE_VALUE(marker.offset);
36
37 HELIOS_ASSERT(marker.block != nullptr, "marker block cannot be null");
38 auto* const target = static_cast<Block*>(marker.block);
39
40 Block* current = head_.load(std::memory_order_acquire);
41 while (current != nullptr && current != target) {
42 Block* const next = current->next.load(std::memory_order_relaxed);
43 const size_t capacity = current->capacity;
44 HELIOS_MEMORY_PROFILE_FREE(current, "StackAllocator");
45 std::destroy_at(current);
46 AlignedFree(current, false);
47 total_capacity_.fetch_sub(capacity, std::memory_order_relaxed);
48 block_count_.fetch_sub(1, std::memory_order_relaxed);
49 current = next;
50 }
51
52 HELIOS_ASSERT(current == target, "Marker block not found in stack chain!");
53
54 target->offset.store(marker.offset, std::memory_order_release);
55 head_.store(target, std::memory_order_release);
56
57 allocation_count_.store(0, std::memory_order_relaxed);
58 total_allocated_.store(marker.offset, std::memory_order_relaxed);
59}

61auto StackAllocator::CreateBlock(size_t capacity) noexcept -> Block* {
62 constexpr size_t kHeader = AlignUp(sizeof(Block), kDefaultAlignment);
63 const size_t total_size = SaturatingAdd(kHeader, capacity);
64 void* const raw = AlignedAlloc(kDefaultAlignment, total_size, false);
65 if (raw == nullptr) [[unlikely]] {
66 return nullptr;
67 }
68
69 HELIOS_MEMORY_PROFILE_ALLOC(raw, total_size, "StackAllocator");
70
71 auto* const block = std::construct_at(static_cast<Block*>(raw));
72 block->buffer = static_cast<std::byte*>(raw) + kHeader;
73 block->capacity = capacity;
74 block->offset.store(0, std::memory_order_relaxed);
75 block->next.store(nullptr, std::memory_order_relaxed);
76 return block;
77}

94auto StackAllocator::TryReserve(Block& block, size_t size,
95 size_t alignment) noexcept -> Reservation {
96 constexpr size_t kHeaderSize = sizeof(AllocationHeader);
97 const size_t header_alignment =
98 std::max(alignment, alignof(AllocationHeader));
99
100 size_t current_offset = block.offset.load(std::memory_order_acquire);
101 for (;;) {
102 auto* const start = static_cast<std::byte*>(block.buffer) + current_offset;
103 const size_t padding =
104 CalculatePaddingWithHeader(start, header_alignment, kHeaderSize);
105 const size_t consumed = SaturatingAdd(padding, size);
106 const size_t end_offset = SaturatingAdd(current_offset, consumed);
107
108 if (end_offset > block.capacity) {
109 return {};
110 }
111
112 if (block.offset.compare_exchange_weak(current_offset, end_offset,
113 std::memory_order_release,
114 std::memory_order_acquire)) {
115 auto* const user_ptr = start + padding;
116 auto* const header_ptr = std::launder(
117 reinterpret_cast<AllocationHeader*>(user_ptr - kHeaderSize));
118 header_ptr->previous_offset = current_offset;
119 header_ptr->total_size = consumed;
120
121 return {
122 .ptr = user_ptr,
123 .allocation_size = size,
124 .consumed = consumed,
125 .previous_offset = current_offset,
126 };
127 }
128 }

131bool StackAllocator::EnsureCapacity(size_t min_capacity) noexcept {
132 Block* observed_head = head_.load(std::memory_order_acquire);
133 const size_t current_capacity =
134 observed_head != nullptr ? observed_head->capacity : initial_capacity_;
135 const size_t desired_capacity =
136 growth_.NextCapacity(current_capacity, min_capacity);
137 if (desired_capacity < min_capacity || desired_capacity == 0) {
138 return false;
139 }
140
141 auto expected = GrowState::kIdle;
142 if (!grow_state_.compare_exchange_strong(expected, GrowState::kGrowing,
143 std::memory_order_acq_rel,
144 std::memory_order_acquire)) {
145 grow_state_.wait(GrowState::kGrowing, std::memory_order_acquire);
146 Block* const new_head = head_.load(std::memory_order_acquire);
147 return new_head != nullptr && new_head->capacity >= min_capacity;
148 }
149
150 Block* const new_block = CreateBlock(desired_capacity);
151 if (new_block == nullptr) {
152 grow_state_.store(GrowState::kIdle, std::memory_order_release);
153 grow_state_.notify_all();
154 return false;
155 }
156
157 PublishBlock(new_block);
158 grow_state_.store(GrowState::kIdle, std::memory_order_release);
159 grow_state_.notify_all();
160 return true;
161}

163void StackAllocator::PublishBlock(Block* block) noexcept {
164 Block* expected_head = head_.load(std::memory_order_acquire);
165 do {
166 block->next.store(expected_head, std::memory_order_relaxed);
167 } while (!head_.compare_exchange_weak(expected_head, block,
168 std::memory_order_release,
169 std::memory_order_acquire));
170
171 total_capacity_.fetch_add(block->capacity, std::memory_order_relaxed);
172 block_count_.fetch_add(1, std::memory_order_relaxed);
173}

175void* StackAllocator::do_allocate(size_t bytes, size_t alignment) {
176 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::StackAllocator::do_allocate");
177 HELIOS_MEMORY_PROFILE_ZONE_VALUE(bytes);
178
179 if (bytes == 0) [[unlikely]] {
180 return nullptr;
181 }
182
183 HELIOS_VERIFY(IsPowerOfTwo(alignment),
184 "alignment must be a power of two, got {}!", alignment);
185
186 const size_t effective_alignment = std::max(alignment, kMinAlignment);
187 Block* head = head_.load(std::memory_order_acquire);
188
189 Reservation reservation{};
190 if (head != nullptr) {
191 reservation = TryReserve(*head, bytes, effective_alignment);
192 }
193
194 if (reservation.ptr == nullptr) {
195 constexpr size_t kHeader = sizeof(AllocationHeader);
196 const size_t min_capacity =
197 SaturatingAdd(SaturatingAdd(bytes, effective_alignment), kHeader);
198 HELIOS_VERIFY(EnsureCapacity(min_capacity),
199 "Unable to grow stack allocator block chain!");
200
201 head = head_.load(std::memory_order_acquire);
202 HELIOS_ASSERT(head != nullptr, "Head block must exist after grow!");
203 reservation = TryReserve(*head, bytes, effective_alignment);
204 HELIOS_VERIFY(reservation.ptr != nullptr,
205 "Reservation failed after successful stack grow!");
206 }
207
208 allocation_count_.fetch_add(1, std::memory_order_relaxed);
209 total_allocations_.fetch_add(1, std::memory_order_relaxed);
210 total_allocated_.fetch_add(reservation.consumed, std::memory_order_relaxed);
211 alignment_waste_.fetch_add(reservation.consumed - reservation.allocation_size,
212 std::memory_order_relaxed);
213 const size_t total = total_allocated_.load(std::memory_order_relaxed);
214 AccumulatePeak(peak_usage_, total);
215
216 return reservation.ptr;
217}

219void StackAllocator::do_deallocate(void* ptr, size_t bytes,
220 size_t /*alignment*/) {
221 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::StackAllocator::do_deallocate");
222 HELIOS_MEMORY_PROFILE_ZONE_VALUE(bytes);
223
224 if (ptr == nullptr) [[unlikely]] {
225 return;
226 }
227
228 Block* const head = head_.load(std::memory_order_acquire);
229 if (head == nullptr) {
230 return;
231 }
232
233 const auto addr = reinterpret_cast<uintptr_t>(ptr);
234 const auto begin = reinterpret_cast<uintptr_t>(head->buffer);
235 const auto end = begin + head->capacity;
236 if (addr < begin || addr >= end) {
237 return;
238 }
239
240 auto* const header = std::launder(reinterpret_cast<AllocationHeader*>(
241 static_cast<std::byte*>(ptr) - sizeof(AllocationHeader)));
242 const size_t current_offset = head->offset.load(std::memory_order_acquire);
243
244 const size_t expected_end =
245 SaturatingAdd(header->previous_offset, header->total_size);
246 if (expected_end != current_offset) {
247 return;
248 }
249
250 head->offset.store(header->previous_offset, std::memory_order_release);
251 allocation_count_.fetch_sub(1, std::memory_order_relaxed);
252 total_deallocations_.fetch_add(1, std::memory_order_relaxed);
253 total_allocated_.fetch_sub(header->total_size, std::memory_order_relaxed);
254
255 if (bytes > 0) {
256 const size_t waste =
257 header->total_size > bytes ? header->total_size - bytes : 0;
258 alignment_waste_.fetch_sub(
259 std::min(waste, alignment_waste_.load(std::memory_order_relaxed)),
260 std::memory_order_relaxed);
261 }
262}
