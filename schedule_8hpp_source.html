42/**
43 * @brief Concept for schedules.
44 * @details A schedule type must be an object and be empty.
45 */
46template <typename T>
47concept ScheduleTrait = std::is_object_v<std::remove_cvref_t<T>> &&
48 std::is_empty_v<std::remove_cvref_t<T>>;
49
50/**
51 * @brief Concept for schedules that provide a name.
52 * @details A schedule with name trait must satisfy `ScheduleTrait` and provide:
53 * - `static constexpr std::string_view kName` variable
54 */
55template <typename T>
56concept ScheduleWithNameTrait = ScheduleTrait<T> && requires {
57 { std::remove_cvref_t<T>::kName } -> std::convertible_to<std::string_view>;
58};
59
60/**
61 * @brief Schedule name for debugging and serialization.
62 * @tparam T Schedule type
63 * @param instance Schedule instance
64 * @return Schedule name
65 */
66template <ScheduleTrait T>
67[[nodiscard]] constexpr std::string_view ScheduleNameOf(
68 const T& /*instance*/ = {}) noexcept {
69 if constexpr (ScheduleWithNameTrait<T>) {
70 return std::remove_cvref_t<T>::kName;
71 } else {
72 return utils::QualifiedTypeNameOf<T>();
73 }
74}

142 * @param vec The vector to append to
143 * @param value The value to append
144 */
145 template <typename T>
146 static void AppendUnique(std::vector<T>& vec, const T& value) {

1062 /**
1063 * @brief Holds the conflict data between systems in the compiled plan.
1064 *
1065 * @details Two representations are maintained:
1066 * - A dense `uint64_t` bitmask per system when `system_count <= 64` (fast
1067 * path, avoids heap indirection on every executor dispatch).
1068 * - A full `vector<vector<bool>>` matrix for larger schedules.
1069 *
1070 * Callers always use `Conflicts(i, j)` and never inspect the fields
1071 * directly, so the switch between representations is invisible at use sites.
1072 */
1073 struct ConflictData {
1074 /// Active when system_count <= 64; bitmask[i] has bit j set iff i and j
1075 /// conflict.
1076 std::vector<uint64_t> bitmask;
1077 /// Active when system_count > 64; empty when bitmask is in use.
1078 std::vector<std::vector<bool>> matrix;
1079
1080 /// Returns true when systems at indices `i` and `j` have a
1081 /// data-access conflict and no explicit ordering constraint between them.
1082 [[nodiscard]] constexpr bool Conflicts(size_t i, size_t j) const noexcept {
1083 if (!bitmask.empty()) {
1084 return static_cast<bool>((bitmask[i] >> j) & 1U);
1085 }
1086 return matrix[i][j];
1087 }
