64void FlamegraphBackend::EndZone(std::span<std::byte> storage) noexcept {
65 auto* state = AsZoneState(storage);
66 const uint64_t end_ts = NowMicros();
67
68 CompleteEvent event;
69 event.name = std::string{state->name};
70 event.start_ts = state->start_ts;
71 event.end_ts = end_ts;
72 event.tid = ThreadId();
73 event.has_text = state->has_text;
74 event.has_value = state->has_value;
75 if (state->has_text) {
76 event.text = std::string{state->text};
77 }
78 if (state->has_value) {
79 event.value = state->value;
80 }
81
82 std::destroy_at(state);
83
84 event_queue_.enqueue(std::move(event));
85 MaybeFlush();
86}

214void FlamegraphBackend::WriteEvent(const FlamegraphEvent& event) noexcept {
215 if (first_event_written_) {
216 output_file_ << ",\n";
217 }
218 first_event_written_ = true;
219
220 std::visit(
221 [this](const auto& e) {
222 using T = std::decay_t<decltype(e)>;
223
224 if constexpr (std::same_as<T, CompleteEvent>) {
225 output_file_ << " {";
226 output_file_ << R"("name":)";
227 WriteJsonString(output_file_, e.name);
228 output_file_ << R"(,"ph":"X","ts":)" << e.start_ts;
229 output_file_ << R"(,"dur":)" << (e.end_ts - e.start_ts);
230 output_file_ << R"(,"pid":0,"tid":)" << e.tid;
231
232 if (e.has_text || e.has_value) {
233 output_file_ << R"(,"args":{)";
234 bool first_arg = true;
235 if (e.has_text) {
236 output_file_ << R"("text":)";
237 WriteJsonString(output_file_, e.text);
238 first_arg = false;
239 }
240 if (e.has_value) {
241 if (!first_arg) {
242 output_file_ << ',';
243 }
244 output_file_ << R"("value":)" << e.value;
245 }
246 output_file_ << '}';
247 }
248
249 output_file_ << '}';
250 } else if constexpr (std::same_as<T, InstantEvent>) {
251 output_file_ << " {";
252 output_file_ << R"("name":)";
253 WriteJsonString(output_file_, e.name);
254 output_file_ << R"(,"ph":"i","ts":)" << e.ts;
255 output_file_ << R"(,"pid":0,"tid":)" << e.tid;
256 output_file_ << R"(,"s":")" << e.scope << '"';
257 output_file_ << '}';
258 } else if constexpr (std::same_as<T, CounterEvent>) {
259 output_file_ << " {";
260 output_file_ << R"("name":)";
261 WriteJsonString(output_file_, e.name);
262 output_file_ << R"(,"ph":"C","ts":)" << e.ts;
263 output_file_ << R"(,"pid":0,"args":{)";
264 WriteJsonString(output_file_, e.name);
265 output_file_ << ':' << e.value;
266 output_file_ << "}}";
267 }
268 },
269 event);
270}

272void FlamegraphBackend::FlushBatch(size_t count) noexcept {
273 if (!file_open_ || count == 0) {
274 return;
275 }
276
277 if (flush_lock_.test_and_set(std::memory_order_acquire)) {
278 return;
279 }
280
281 FlamegraphEvent event{std::in_place_type<CompleteEvent>};
282 size_t written = 0;
283 while (written < count && event_queue_.try_dequeue(event)) {
284 WriteEvent(event);
285 ++written;
286 }
287
288 flush_lock_.clear(std::memory_order_release);
289}
290
291void FlamegraphBackend::FlushAll() noexcept {
292 if (!file_open_) {
293 return;
294 }
295
296 while (flush_lock_.test_and_set(std::memory_order_acquire)) {
297 }
298
299 FlamegraphEvent event{std::in_place_type<CompleteEvent>};
300 while (event_queue_.try_dequeue(event)) {
301 WriteEvent(event);
302 }
303
304 output_file_ << "\n]\n";
305 output_file_.close();
306 file_open_ = false;
307
308 flush_lock_.clear(std::memory_order_release);
309}
