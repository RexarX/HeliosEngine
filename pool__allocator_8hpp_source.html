195inline PoolAllocator::PoolAllocator(PoolAllocatorOptions options) noexcept
196 : block_size_(std::max(options.block_size, sizeof(void*))),
197 initial_block_count_(options.block_count),
198 alignment_(options.alignment),
199 growth_(options.growth) {
200 HELIOS_ASSERT(block_size_ > 0, "block_size must be greater than zero!");
201 HELIOS_ASSERT(initial_block_count_ > 0,
202 "block_count must be greater than zero!");
203 HELIOS_ASSERT(IsPowerOfTwo(alignment_),
204 "alignment '{}' must be power of two!", alignment_);
205 HELIOS_ASSERT(alignment_ >= alignof(void*), "alignment '{}' must be >= '{}'!",
206 alignment_, alignof(void*));
207
208 block_size_ = AlignUp(block_size_, alignment_);
209
210 ChunkHeader* const initial_chunk =
211 CreateChunk(block_size_, initial_block_count_, alignment_);
212 HELIOS_VERIFY(initial_chunk != nullptr, "Failed to allocate pool chunk!");
213
214 chunks_.store(initial_chunk, std::memory_order_release);
215 total_blocks_.store(initial_chunk->block_count, std::memory_order_relaxed);
216 free_blocks_.store(initial_chunk->block_count, std::memory_order_relaxed);
217 PushChunkBlocks(*initial_chunk);
218}

317inline auto PoolAllocator::Stats() const noexcept -> AllocatorStats {
318 const size_t total = total_blocks_.load(std::memory_order_relaxed);
319 const size_t free = free_blocks_.load(std::memory_order_relaxed);
320 const size_t used = total >= free ? total - free : 0;
321 const size_t peak = peak_used_blocks_.load(std::memory_order_relaxed);
322
323 return {
324 .total_allocated = SaturatingMul(used, block_size_),
325 .peak_usage = SaturatingMul(peak, block_size_),
326 .allocation_count = used,
327 .total_allocations = total_allocations_.load(std::memory_order_relaxed),
328 .total_deallocations =
329 total_deallocations_.load(std::memory_order_relaxed),
330 .alignment_waste = 0,
331 };
332}
