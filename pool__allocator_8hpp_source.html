27/**

247inline auto PoolAllocator::Stats() const noexcept -> AllocatorStats {
248 const size_t total = total_blocks_.load(std::memory_order_relaxed);
249 const size_t free = free_blocks_.load(std::memory_order_relaxed);
250 const size_t used = total >= free ? total - free : 0;
251 const size_t peak = peak_used_blocks_.load(std::memory_order_relaxed);
252
253 return {
254 .total_allocated = SaturatingMul(used, block_size_),
255 .peak_usage = SaturatingMul(peak, block_size_),
256 .allocation_count = used,
257 .total_allocations = total_allocations_.load(std::memory_order_relaxed),
258 .total_deallocations =
259 total_deallocations_.load(std::memory_order_relaxed),
260 .alignment_waste = 0,
261 };
262}
