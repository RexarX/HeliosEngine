68/**
69 * @brief Concept for systems that declare all data access via parameter types.
70 * @details A system must be destructible, move or copy constructible, not
71 * polymorphic, an object type, and its `operator()` parameters must all have
72 * `SystemParamTraits` specialisations.
73 */
74template <typename T>
75concept SystemTrait =
76 std::destructible<T> &&
77 (std::move_constructible<T> || std::copy_constructible<T>) &&
78 !std::is_polymorphic_v<std::remove_cvref_t<T>> &&
79 std::is_object_v<std::remove_cvref_t<T>> &&
80 details::HasSystemParamCallOperator<T>;
