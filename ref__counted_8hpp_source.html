28/**
29 * @brief CRTP base class that embeds a non-atomic reference counter into the
30 * derived type.
31 * @details Inherit from this class to make a type usable with `RefCounted<T>`.
32 * The counter tracks the number of `RefCounted` handles pointing to the object.
33 * The counter is NOT thread-safe; use `ArcFromThis` if thread-safety is
34 * required.
35 *
36 * The counter starts at 0.
37 * `RefCounted<T>` increments it on construction and decrements it on
38 * destruction, deleting the object when it reaches 0.
39 * @warning Do not copy or move `RcFromThis` directly;
40 * copies share the same counter slot but should be managed exclusively through
41 * `RefCounted<T>`.
42 * @tparam Derived The concrete class inheriting from this base
43 *
44 * @code
45 * class Mesh final : public helios::mem::RcFromThis<Mesh> {
46 * public:
47 * explicit Mesh(int id) : id_(id) {}
48 * int Id() const { return id_; }
49 * private:
50 * int id_;
51 * };
52 *
53 * auto mesh = helios::mem::MakeRc<Mesh>(42);
54 * @endcode
55 */
56template <typename Derived>
57class RcFromThis {
58public:
59 RcFromThis() noexcept = default;
60 RcFromThis(const RcFromThis& /*other*/) noexcept {}
61 RcFromThis(RcFromThis&& /*other*/) noexcept {}
62 ~RcFromThis() noexcept = default;
63
64 RcFromThis& operator=(const RcFromThis& /*other*/) noexcept { return *this; }
65 RcFromThis& operator=(RcFromThis&& /*other*/) noexcept { return *this; }
66
67 /**
68 * @brief Returns the current reference count.
69 * @return Current number of `RefCounted` handles owning this object
70 */
71 [[nodiscard]] uint32_t RefCount() const noexcept { return ref_count_; }
72
73private:
74 template <typename D, typename A>
75 requires std::derived_from<D, RcFromThis<D>>
76 friend class RefCounted;
77
78 /// @brief Increments the reference count by 1.
79 void AddRef() noexcept { ++ref_count_; }
80
81 /**
82 * @brief Decrements the reference count and checks if it reached zero.
83 * @warning Triggers an assertion if called when the ref count is already
84 * zero.
85 * @return true if the object should be deleted (ref count reached 0)

99/**
100 * @brief CRTP base class that embeds an atomic reference counter into the
101 * derived type.
102 * @details Inherit from this class to make a type usable with
103 * `AtomicRefCounted<T>`. The counter tracks the number of `AtomicRefCounted`
104 * handles pointing to the object. All counter operations use
105 * `std::memory_order_acq_rel` / `std::memory_order_acquire` to ensure correct
106 * visibility across threads.
107 * @note The counter starts at 0. `AtomicRefCounted<T>` increments it on
108 * construction and decrements it on destruction, deleting the object when it
109 * reaches 0.
110 * @warning Do not manage the same raw pointer from multiple threads without
111 * using `AtomicRefCounted<T>` handles exclusively.
112 * @tparam Derived The concrete class inheriting from this base
113 *
114 * @code
115 * class Texture final : public helios::mem::ArcFromThis<Texture> {
116 * public:
117 * explicit Texture(std::string_view name) : name_(name) {}
118 * std::string_view Name() const { return name_; }
119 * private:
120 * std::string name_;
121 * };

123 * auto tex = helios::mem::MakeArc<Texture>("diffuse");
124 * @endcode

169template <typename Derived>
170inline bool ArcFromThis<Derived>::Release() noexcept {
171 HELIOS_ASSERT(ref_count_.load(std::memory_order_relaxed) > 0,
