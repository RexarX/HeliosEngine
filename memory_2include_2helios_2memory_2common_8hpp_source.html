78 [[nodiscard]] constexpr size_t NextCapacity(
79 size_t current_capacity, size_t min_capacity) const noexcept {
80 if (current_capacity >= min_capacity) {
81 return current_capacity;
82 }
83
84 if (mode == GrowthMode::kFixed) {
85 return 0;
86 }
87
88 size_t next = current_capacity;
89 if (next == 0) {
90 next = kMinAlignment;
91 }
92
93 if (mode == GrowthMode::kLinear) {
94 if (linear_step == 0) {
95 return 0;
96 }
97 while (next < min_capacity) {
98 const size_t previous = next;
99 next = SaturatingAdd(next, linear_step);
100 if (next <= previous) {
101 next = std::numeric_limits<size_t>::max();
102 }
103 if (next == std::numeric_limits<size_t>::max()) {
104 break;
105 }
106 }
107 } else {
108 if (geometric_denominator == 0 ||
109 geometric_numerator < geometric_denominator) {
110 return 0;
111 }
112 while (next < min_capacity) {
113 const size_t previous = next;
114 const size_t scaled = SaturatingMul(next, geometric_numerator);
115 next = scaled / geometric_denominator;
116 if (next <= previous) {
117 next = std::numeric_limits<size_t>::max();
118 }
119 if (next == std::numeric_limits<size_t>::max()) {
120 break;
121 }
122 }
123 }
124
125 return std::min(next, max_capacity);
126 }
127};

180/**
181 * @brief Saturating add for size_t.
182 * @param lhs Left operand
183 * @param rhs Right operand
184 * @return lhs + rhs, clamped to `size_t` max on overflow
185 */
186[[nodiscard]] constexpr size_t SaturatingAdd(size_t lhs, size_t rhs) noexcept {
187 constexpr size_t kMax = std::numeric_limits<size_t>::max();
188 if (rhs > kMax - lhs) {
189 return kMax;
190 }
191 return lhs + rhs;
192}

194/**
195 * @brief Saturating multiply for size_t.
196 * @param lhs Left operand
197 * @param rhs Right operand
198 * @return lhs * rhs, clamped to `size_t` max on overflow
199 */
200[[nodiscard]] constexpr size_t SaturatingMul(size_t lhs, size_t rhs) noexcept {
201 constexpr size_t kMax = std::numeric_limits<size_t>::max();
202 if (lhs == 0 || rhs == 0) {
203 return 0;
204 }
205 if (lhs > kMax / rhs) {
206 return kMax;
207 }
208 return lhs * rhs;
209}

211/**
212 * @brief Aligns value up to alignment.
213 * @param value Value to align
214 * @param alignment Alignment, must be a power of two
215 * @return Aligned value
216 */
217[[nodiscard]] constexpr size_t AlignUp(size_t value,
218 size_t alignment) noexcept {
219 HELIOS_ASSERT(IsPowerOfTwo(alignment), "alignment '{}' must be power of two!",
220 alignment);
221 const size_t mask = alignment - 1;
222 if (value > std::numeric_limits<size_t>::max() - mask) {
223 return std::numeric_limits<size_t>::max();
224 }
225 return (value + mask) & ~mask;
226}

229 * @brief Aligns pointer up to alignment.
230 * @param ptr Pointer to align
231 * @param alignment Alignment, must be a power of two
232 * @return Aligned pointer

234[[nodiscard]] inline void* AlignUpPtr(void* ptr, size_t alignment) noexcept {

236 alignment);
237 auto* const raw = static_cast<std::byte*>(ptr);
238 const auto addr = reinterpret_cast<uintptr_t>(raw);

257/**
258 * @brief Computes padding required for pointer alignment.
259 * @param ptr Current pointer
260 * @param alignment Requested alignment
261 * @return Padding in bytes
262 */
263[[nodiscard]] inline size_t CalculatePadding(const void* ptr,
264 size_t alignment) noexcept {
265 const auto addr = reinterpret_cast<uintptr_t>(ptr);

270/**
271 * @brief Computes padding for aligned storage with a prepended header.
272 * @param ptr Current pointer
273 * @param alignment Requested alignment
274 * @param header_size Header size stored before aligned user memory
275 * @return Padding in bytes, including header space
276 */
277[[nodiscard]] inline size_t CalculatePaddingWithHeader(
278 const void* ptr, size_t alignment, size_t header_size) noexcept {
279 size_t padding = CalculatePadding(ptr, alignment);
280 if (padding < header_size) {
281 padding = SaturatingAdd(padding, AlignUp(header_size - padding, alignment));
282 }
