27 */

29 bool enable_profile = true) noexcept {
30 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::AlignedAlloc");
31 HELIOS_MEMORY_PROFILE_ZONE_VALUE(size);
32 HELIOS_ASSERT(alignment != 0, "alignment cannot be zero!");
33
34 const auto is_power_of_two = [](size_t value) {
35 return value != 0 && (value & (value - 1)) == 0;
36 };
37 HELIOS_ASSERT(is_power_of_two(alignment),
38 "alignment must be a power of two!");
39 HELIOS_ASSERT(size != 0, "size cannot be zero!");

41#ifdef _MSC_VER
