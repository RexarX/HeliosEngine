23/**
24 * @brief PMR arena allocator with lock-free hot allocation path.
25 * @details Allocates memory from chained blocks using atomic bump pointers.
26 *
27 * Allocation hot path is lock-free and performs one CAS on the current head
28 * block. When a block is exhausted, allocator enters a coordinated grow slow
29 * path where a single thread allocates and publishes a new block.
30 *
31 * Individual deallocation is a no-op, matching monotonic arena semantics.
32 * Memory is reclaimed by `Reset()` or destruction.
33 *
34 * @note Thread-safe for `allocate` and `deallocate`.
35 * @warning `Reset()` must not run concurrently with allocation/deallocation.
36 */

168inline ArenaAllocator::ArenaAllocator(ArenaOptions options) noexcept
169 : initial_capacity_(options.initial_capacity), growth_(options.growth) {
170 HELIOS_ASSERT(initial_capacity_ > 0,
171 "initial_capacity must be greater than zero!");
172 HELIOS_ASSERT(growth_.max_capacity >= initial_capacity_,
173 "max_capacity '{}' must be >= initial_capacity '{}'!",
174 growth_.max_capacity, initial_capacity_);
175
176 Block* const initial_block = CreateBlock(initial_capacity_);
177 HELIOS_VERIFY(initial_block != nullptr, "Failed to allocate initial block!");
178
179 head_.store(initial_block, std::memory_order_release);
180 total_capacity_.store(initial_capacity_, std::memory_order_relaxed);
181 block_count_.store(1, std::memory_order_relaxed);
182}
