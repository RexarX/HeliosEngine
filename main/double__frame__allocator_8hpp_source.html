19/**
20 * @brief Double-buffered frame allocator.
21 * @details Maintains two frame buffers, allowing memory from the previous frame
22 * to remain valid while allocating for the current frame.
23 * Useful when data needs to be accessible for one additional frame (e.g., GPU upload buffers, interpolation).
24 *
25 * The allocator automatically switches between buffers on each frame.
26 *
27 * Uses atomic for buffer index and mutex only for frame transitions.
28 * Previous frame's data remains valid until the next frame begins.
29 * Allocations are lock-free since FrameAllocator uses atomic operations internally.
30 *
31 * @note Thread-safe.
32 */
33class DoubleFrameAllocator final {
34public:
35 static constexpr size_t kBufferCount = 2;

43 : allocators_{FrameAllocator(capacity_per_buffer), FrameAllocator(capacity_per_buffer)} {}

175inline DoubleFrameAllocator& DoubleFrameAllocator::operator=(DoubleFrameAllocator&& other) noexcept {
176 if (this == &other) [[unlikely]] {
177 return *this;
178 }
179
180 const std::scoped_lock lock(mutex_);
181 const std::scoped_lock other_lock(other.mutex_);
182
183 allocators_ = std::move(other.allocators_);
184 current_buffer_.store(other.current_buffer_.load(std::memory_order_acquire), std::memory_order_release);

190inline AllocationResult DoubleFrameAllocator::Allocate(size_t size, size_t alignment) noexcept {
191 // Lock-free: current_buffer_ is atomic, FrameAllocator::Allocate is thread-safe via atomics
192 const size_t buffer = current_buffer_.load(std::memory_order_acquire);
193 return allocators_[buffer].Allocate(size, alignment);
194}

196inline void DoubleFrameAllocator::NextFrame() noexcept {
197 const std::scoped_lock lock(mutex_);
198 // Switch to the other buffer
199 const size_t new_buffer = 1 - current_buffer_.load(std::memory_order_relaxed);
200 // Reset the new current buffer before switching
201 allocators_[new_buffer].Reset();
202 // Atomically switch to new buffer
203 current_buffer_.store(new_buffer, std::memory_order_release);
204}
