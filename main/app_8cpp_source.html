73void App::Update() {
74 // Clean up completed per-sub-app overlapping futures
75 for (auto& [index, futures] : sub_app_overlapping_futures_) {
76 std::erase_if(futures,
77 [](auto& future) { return future.wait_for(std::chrono::seconds(0)) == std::future_status::ready; });
78 }
79
80 // Update main sub-app (synchronous)
81 main_sub_app_.Update(executor_);
82
83 // Process non-overlapping sub-apps
84 auto non_overlapping_future = executor_.Run(update_graph_);
85
86 // Launch overlapping sub-apps asynchronously (non-blocking)
87 for (size_t i = 0; i < sub_apps_.size(); ++i) {
88 auto& sub_app = sub_apps_[i];
89 if (!sub_app.AllowsOverlappingUpdates()) {
90 continue;
91 }
92
93 // Capture by reference is safe: App lifetime exceeds sub-app updates
94 auto future = executor_.Async([this, &sub_app]() {
95 sub_app.Extract(main_sub_app_.GetWorld());
96 sub_app.Update(executor_);
97 });
98
99 // Store as shared_future for per-sub-app tracking (allows multiple references)
100 sub_app_overlapping_futures_[i].push_back(future.share());
101 }
102
103 non_overlapping_future.Wait();
104}

106void App::Initialize() {
107 HELIOS_ASSERT(!IsInitialized(), "Failed to initialize: App is already initialized!");
108 HELIOS_ASSERT(!IsRunning(), "Failed to initialize: Cannot initialize while app is running!");
109
110 async::TaskGraph initialization_graph;
111 initialization_graph.EmplaceTask([this]() {
112 main_sub_app_.BuildScheduler();
113 main_sub_app_.ExecuteStage<helios::app::StartUpStage>(executor_);
114 main_sub_app_.GetWorld().Update();
115 });
116
117 initialization_graph.ForEach(sub_apps_, [this](const SubApp& sub_app_ref) mutable {
118 // Need non-const access to modify sub_app
119 auto& sub_app = const_cast<SubApp&>(sub_app_ref);
120
121 sub_app.BuildScheduler();
122 sub_app.ExecuteStage<helios::app::StartUpStage>(executor_);
123 sub_app.GetWorld().Update();
124 });
125
126 auto future = executor_.Run(initialization_graph);
127
128 update_graph_.ForEach(sub_apps_, [this](const SubApp& sub_app_ref) mutable {
129 if (sub_app_ref.AllowsOverlappingUpdates()) {
130 return;
131 }
132
133 // Need non-const access to modify sub_app
134 auto& sub_app = const_cast<SubApp&>(sub_app_ref);
135 sub_app.Extract(main_sub_app_.GetWorld());
136 sub_app.Update(executor_);
137 });
138
139 future.Wait();

173void App::WaitForOverlappingUpdates(const SubApp& sub_app) {
174 // Find the sub-app index by comparing addresses
175 auto sub_app_index = static_cast<size_t>(-1);
176 for (size_t i = 0; i < sub_apps_.size(); ++i) {
177 if (&sub_apps_[i] == &sub_app) {
178 sub_app_index = i;
179 break;
180 }
181 }
182
183 if (sub_app_index == static_cast<size_t>(-1)) [[unlikely]] {
184 return; // Sub-app not found
185 }
186
187 const auto futures_it = sub_app_overlapping_futures_.find(sub_app_index);
188 if (futures_it == sub_app_overlapping_futures_.end()) {
189 return; // No overlapping futures for this sub-app
190 }
191
192 for (auto& future : futures_it->second) {
193 if (future.wait_for(std::chrono::seconds(0)) != std::future_status::ready) {
194 future.wait();
195 }
196 }
197 futures_it->second.clear();
198}

222AppExitCode App::DefaultRunner(App& app) {
223 // Simple loop that runs until interrupted
224 // Users should provide their own runner for more complex behavior
225 try {
226 while (app.IsRunning()) {
227 app.Update();
228 }
