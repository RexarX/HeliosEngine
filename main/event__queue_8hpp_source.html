150inline void EventQueue::Register() {

152 if (!storages_.contains(type_id)) {

197inline void EventQueue::Merge(EventQueue& other) {
198 for (auto& [type_id, other_storage] : other.storages_) {
199 if (other_storage.Empty()) {
200 continue;
201 }
202
203 if (const auto it = storages_.find(type_id); it == storages_.end()) {
204 // If we don't have this type registered, move the entire storage
205 storages_.emplace(type_id, std::move(other_storage));
206 } else {
207 auto& this_storage = it->second;
208 if (this_storage.Empty()) {
209 // If our storage is empty, just move the other storage
210 this_storage = std::move(other_storage);
211 } else {
212 // Append raw bytes from other storage to this storage
213 this_storage.AppendRawBytes(other_storage.Data());
214 }
215 }
216 }
