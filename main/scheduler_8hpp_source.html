476template <ScheduleTrait S>
477inline void Scheduler::RegisterSchedule() {
478 constexpr ScheduleId schedule_id = ScheduleIdOf<S>();
479 if (schedules_.contains(schedule_id)) [[unlikely]] {
480 return; // Already registered
481 }
482
483 schedules_.emplace(schedule_id, ScheduleExecutor(schedule_id));
484
485 // Store schedule ordering constraints
486 ScheduleOrdering ordering;
487
488 // Extract Before constraints from schedule type
489 if constexpr (ScheduleWithBeforeTrait<S>) {
490 constexpr auto before = ScheduleBeforeOf<S>();
491 ordering.before.reserve(before.size());
492 for (const ScheduleId id : before) {
493 ordering.before.push_back(id);
494 }
495 }
496
497 // Extract After constraints from schedule type
498 if constexpr (ScheduleWithAfterTrait<S>) {
499 constexpr auto after = ScheduleAfterOf<S>();
500 ordering.after.reserve(after.size());
501 for (const ScheduleId id : after) {
502 ordering.after.push_back(id);
503 }
504 }
505
506 // Store stage membership
507 ordering.stage_id = ScheduleStageOf<S>();
508
509 schedule_constraints_[schedule_id] = std::move(ordering);
510}

578inline void Scheduler::ExecuteStage(ecs::World& world, async::Executor& executor) {
579 constexpr ScheduleId stage_id = ScheduleIdOf<S>();
580
581 // Execute all schedules that belong to this stage in topological order
582 for (const ScheduleId schedule_id : schedule_order_) {
583 const auto constraint_it = schedule_constraints_.find(schedule_id);
584 if (constraint_it == schedule_constraints_.end()) {
585 continue;
586 }

590 ExecuteScheduleById(schedule_id, world, executor);

616inline void Scheduler::AddSetRunsBefore(SystemSetId before_id, SystemSetId after_id) {
617 if (before_id == after_id) [[unlikely]] {
618 return;
619 }
620
621 auto before_it = system_sets_.find(before_id);
622 if (before_it == system_sets_.end()) {
623 // Set may not have been explicitly registered yet; create a minimal entry.
624 SystemSetInfo info{};
625 info.id = before_id;
626 before_it = system_sets_.emplace(before_id, std::move(info)).first;
627 }
628
629 auto after_it = system_sets_.find(after_id);
630 if (after_it == system_sets_.end()) {
631 SystemSetInfo info{};
632 info.id = after_id;
633 after_it = system_sets_.emplace(after_id, std::move(info)).first;
634 }
635
636 auto& before_info = before_it->second;
637 auto& after_info = after_it->second;
638
639 // Encode relationship:
640 // - before_info.before_sets contains after_id (before runs before after)
641 // - after_info.after_sets contains before_id
642 if (std::ranges::find(before_info.before_sets, after_id) == before_info.before_sets.end()) {
643 before_info.before_sets.push_back(after_id);
644 }
645 if (std::ranges::find(after_info.after_sets, before_id) == after_info.after_sets.end()) {
646 after_info.after_sets.push_back(before_id);
647 }
648}
