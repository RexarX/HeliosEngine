20/**
21 * @brief Stack/linear allocator with LIFO deallocation support.
22 * @details Allocates memory sequentially using a bump pointer, but unlike
23 * FrameAllocator, supports LIFO (stack-like) deallocations. Each allocation
24 * stores a header with the previous offset, allowing proper unwinding.
25 *
26 * Ideal for hierarchical/scoped allocations where deallocation follows
27 * allocation order (e.g., call stacks, recursive algorithms).
28 *
29 * Each allocation has a small header overhead for tracking.
30 *
31 * Uses mutex for allocation/deallocation operations.
32 *
33 * @note Thread-safe.
34 * @warning Deallocations must follow LIFO order (stack discipline).
35 */

300inline AllocationResult StackAllocator::Allocate(size_t size, size_t alignment) noexcept {
301 HELIOS_ASSERT(IsPowerOfTwo(alignment), "Failed to allocate memory: alignment must be power of 2, got '{}'!",
302 alignment);
303 HELIOS_ASSERT(alignment >= kMinAlignment, "Failed to allocate memory: alignment must be at least '{}', got '{}'!",
304 kMinAlignment, alignment);
305
306 if (size == 0) [[unlikely]] {
307 return {.ptr = nullptr, .allocated_size = 0};
308 }
309
310 const std::scoped_lock lock(mutex_);
311
312 const size_t current_offset = offset_.load(std::memory_order_relaxed);
313
314 // Calculate space needed for header + alignment
315 auto* current_ptr = static_cast<uint8_t*>(buffer_) + current_offset;
316 const size_t header_padding = CalculatePaddingWithHeader(current_ptr, alignment, sizeof(AllocationHeader));
317 const size_t required_space = header_padding + size;
318
319 // Check if we have enough space
320 if (current_offset + required_space > capacity_) {
321 return {.ptr = nullptr, .allocated_size = 0};
322 }
323
324 // Store header at the padded position
325 auto* header_ptr = current_ptr + header_padding - sizeof(AllocationHeader);
326 auto* header = reinterpret_cast<AllocationHeader*>(header_ptr);
327 header->previous_offset = current_offset;
328 header->padding = header_padding;
329
330 // Calculate aligned data pointer
331 auto* data_ptr = current_ptr + header_padding;
332
333 // Update offset
334 const size_t new_offset = current_offset + required_space;
335 offset_.store(new_offset, std::memory_order_release);
336
337 // Update stats
338 allocation_count_.fetch_add(1, std::memory_order_relaxed);
339 total_allocations_.fetch_add(1, std::memory_order_relaxed);
340 alignment_waste_.fetch_add(header_padding - sizeof(AllocationHeader), std::memory_order_relaxed);
341
342 // Update peak offset
343 size_t current_peak = peak_offset_.load(std::memory_order_acquire);
344 while (new_offset > current_peak) {
345 if (peak_offset_.compare_exchange_weak(current_peak, new_offset, std::memory_order_release,
346 std::memory_order_acquire)) {
347 break;
348 }
349 }
350
351 return {data_ptr, size};
352}

354inline void StackAllocator::Deallocate(void* ptr, [[maybe_unused]] size_t size) noexcept {
355 if (ptr == nullptr) [[unlikely]] {
356 return;
357 }
358
359 HELIOS_ASSERT(Owns(ptr), "Failed to deallocate memory: ptr does not belong to this allocator!");
360
361 const std::scoped_lock lock(mutex_);
362
363 // Get the header stored before the allocation
364 auto* header = reinterpret_cast<AllocationHeader*>(static_cast<uint8_t*>(ptr) - sizeof(AllocationHeader));
365
366 // Verify this is the most recent allocation (LIFO check)
367 [[maybe_unused]] const size_t current_offset = offset_.load(std::memory_order_relaxed);
368 HELIOS_ASSERT(static_cast<uint8_t*>(ptr) + size <= static_cast<uint8_t*>(buffer_) + current_offset,
369 "Failed to deallocate memory: Deallocation violates LIFO order!");
370
371 // Rewind to previous offset
372 offset_.store(header->previous_offset, std::memory_order_release);
373
374 // Update stats
375 allocation_count_.fetch_sub(1, std::memory_order_relaxed);
376 total_deallocations_.fetch_add(1, std::memory_order_relaxed);
377}

386inline void StackAllocator::RewindToMarker(size_t marker) noexcept {
387 const std::scoped_lock lock(mutex_);
388
389 [[maybe_unused]] const size_t current_offset = offset_.load(std::memory_order_relaxed);
390 HELIOS_ASSERT(marker <= current_offset, "Failed to rewind to marker: marker '{}' is ahead of current offset '{}'!",
391 marker, current_offset);
392 HELIOS_ASSERT(marker <= capacity_, "Failed to rewind to marker: marker '{}' exceeds capacity '{}'!", marker,
393 capacity_);
394
395 offset_.store(marker, std::memory_order_release);
396
397 // Note: allocation_count_ becomes inaccurate after rewind, but that's acceptable
398 // since we're doing bulk deallocation
399}

401inline bool StackAllocator::Owns(const void* ptr) const noexcept {
402 if (ptr == nullptr || buffer_ == nullptr) {
403 return false;
404 }
405
406 const auto addr = reinterpret_cast<uintptr_t>(ptr);
407 const auto start = reinterpret_cast<uintptr_t>(buffer_);
408 const auto end = start + capacity_;
409
410 return addr >= start && addr < end;
411}
