235inline auto DynamicLibrary::Reload() -> std::expected<void, DynamicLibraryError> {
236 if (!Loaded()) {
237 return std::unexpected(DynamicLibraryError::NotLoaded);
238 }
239
240 const auto saved_path = path_;
241
242 auto unload_result = Unload();
243 if (!unload_result) {
244 return unload_result;
245 }
246
247 return Load(saved_path);
248}
