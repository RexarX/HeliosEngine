122inline void CmdQueue::EnqueueBulk(R&& commands) {
123#ifdef HELIOS_ENABLE_ASSERTS

130 // Reserve space to avoid multiple reallocations
131 commands_.reserve(commands_.size() + std::ranges::size(commands));
132 }
