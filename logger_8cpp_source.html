35[[nodiscard]] auto GenerateTimestamp() noexcept
36 -> std::expected<Timestamp, std::string_view> {
37 try {
38 const auto now = std::chrono::system_clock::now();
39 const auto time_t_now = std::chrono::system_clock::to_time_t(now);
40
41 std::tm local_time{};
42
43 // Needed because of compatibility issues on Windows
44 // For more info: https://en.cppreference.com/w/c/chrono/localtime.html
45#ifdef HELIOS_PLATFORM_WINDOWS
46 const errno_t err = localtime_s(&local_time, &time_t_now);
47 if (err != 0) [[unlikely]] {
48 return std::unexpected("Failed to get local time");
49 }
50#else
51 const std::tm* const local_time_ptr = localtime_r(&time_t_now, &local_time);
52 if (local_time_ptr == nullptr) [[unlikely]] {
53 return std::unexpected("Failed to get local time");
54 }
55#endif
56
57 const int year = local_time.tm_year + 1900;
58 const int month = local_time.tm_mon + 1;
59 const int day = local_time.tm_mday;
60 const int hour = local_time.tm_hour;
61 const int min = local_time.tm_min;
62 const int sec = local_time.tm_sec;
63
64 std::array<char, 20> time_buf = {};
65 const auto result = std::format_to(
66 time_buf.begin(), "{:04d}-{:02d}-{:02d}_{:02d}-{:02d}-{:02d}", year,
67 month, day, hour, min, sec);
68 const auto len =
69 static_cast<size_t>(std::distance(time_buf.begin(), result));
70 return Timestamp(std::string_view(time_buf.data(), len));
71 } catch (...) {
72 return std::unexpected("Exception during timestamp generation");
73 }
74}

76[[nodiscard]] std::string FormatLogFileName(std::string_view logger_name,
77 std::string_view pattern) {
78 std::string result(pattern);
79
80 size_t pos = result.find("{name}");
81 if (pos != std::string::npos) {
82 result.replace(pos, 6, logger_name);
83 }
84
85 pos = result.find("{timestamp}");
86 if (pos != std::string::npos) {
87 const Timestamp timestamp = GenerateTimestamp().value_or("unknown_time");
88 result.replace(pos, 11, timestamp.View());
89 }
90
91 return result;
92}

209auto Logger::CreateLogger(std::string_view logger_name,
210 const Config& config) noexcept
211 -> std::shared_ptr<spdlog::logger> {
212 try {
213 std::vector<std::shared_ptr<spdlog::sinks::sink>> log_sinks;
214 log_sinks.reserve(2);
215
216 // Create file sink if enabled
217 if (config.enable_file) {
218 // Create log directory if it doesn't exist
219 std::error_code ec;
220 std::filesystem::create_directories(config.log_directory, ec);
221 if (ec) {
222 // Failed to create directory, continue without file logging
223 } else {
224 // Format the log file name
225 const std::string log_file_name =
226 FormatLogFileName(logger_name, config.file_name_pattern);
227 std::filesystem::path log_file_path(config.log_directory);
228 log_file_path /= log_file_name;
229
230 auto file_formatter = std::make_unique<spdlog::pattern_formatter>();
231 file_formatter->set_pattern(std::string(config.file_pattern));
232 auto file_sink = std::make_shared<spdlog::sinks::basic_file_sink_mt>(
233 log_file_path.string(), config.truncate_files);
234 file_sink->set_formatter(std::move(file_formatter));
235 log_sinks.push_back(std::move(file_sink));
236 }
237 }
238
239 // Create console sink if enabled
240 if (config.enable_console) {
241 auto console_formatter = std::make_unique<spdlog::pattern_formatter>();
242 console_formatter->set_pattern(std::string(config.console_pattern));
243 auto console_sink =
244 std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
245 console_sink->set_formatter(std::move(console_formatter));
246 log_sinks.emplace_back(std::move(console_sink));
247 }
248
249 if (log_sinks.empty()) [[unlikely]] {
250 return nullptr;
251 }
252
253 std::shared_ptr<spdlog::logger> logger;
254 if (config.async_logging) {
255 // Initialize spdlog's global thread pool once per process
256 static std::once_flag s_init_flag;
257 std::call_once(s_init_flag, [&] {
258 spdlog::init_thread_pool(config.async_queue_size,
259 config.async_thread_count);
260 });
261
262 logger = std::make_shared<spdlog::async_logger>(
263 std::string(logger_name), std::make_move_iterator(log_sinks.begin()),
264 std::make_move_iterator(log_sinks.end()), spdlog::thread_pool(),
265 spdlog::async_overflow_policy::block);
266 } else {
267 logger = std::make_shared<spdlog::logger>(
268 std::string(logger_name), std::make_move_iterator(log_sinks.begin()),
269 std::make_move_iterator(log_sinks.end()));
270 }
271
272 spdlog::register_logger(logger);
273 logger->set_level(spdlog::level::trace);
274 logger->flush_on(static_cast<spdlog::level::level_enum>(
275 std::to_underlying(config.auto_flush_level)));
276 return logger;
277 } catch (...) {
278 return nullptr;
279 }
280}
