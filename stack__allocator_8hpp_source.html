21/**
22 * @brief PMR stack allocator with LIFO-aware deallocation.
23 * @details Uses block-chained bump allocation with per-allocation headers.
24 *
25 * Allocation path is lock-free with atomic CAS on head block offset. Deallocate
26 * tries LIFO rewind when possible. Non-LIFO deallocation is treated as no-op,
27 * which keeps PMR compatibility for containers that may free out of order.
28 *
29 * @note Thread-safe for `allocate`. `Reset()` and marker operations require
30 * external synchronization.
31 */

180inline StackAllocator::StackAllocator(StackAllocatorOptions options) noexcept
181 : initial_capacity_(options.initial_capacity), growth_(options.growth) {
182 HELIOS_ASSERT(initial_capacity_ > 0,
183 "initial_capacity must be greater than zero!");
184 HELIOS_ASSERT(growth_.max_capacity >= initial_capacity_,
185 "max_capacity '{}' must be >= initial_capacity '{}'!",
186 growth_.max_capacity, initial_capacity_);
187
188 Block* const initial_block = CreateBlock(initial_capacity_);
189 HELIOS_VERIFY(initial_block != nullptr, "Failed to allocate initial block!");
190 head_.store(initial_block, std::memory_order_release);
191 total_capacity_.store(initial_capacity_, std::memory_order_relaxed);
192 block_count_.store(1, std::memory_order_relaxed);
193}

288inline void StackAllocator::Reset() noexcept {
289 HELIOS_MEMORY_PROFILE_SCOPE_N("helios::mem::StackAllocator::Reset");
290
291 Block* const head = head_.load(std::memory_order_acquire);
292 if (head == nullptr) {
293 return;
294 }
295
296 FreeChain(head);
297 Block* const fresh = CreateBlock(initial_capacity_);
298 HELIOS_VERIFY(fresh != nullptr, "Failed to allocate block during reset!");
299
300 head_.store(fresh, std::memory_order_release);
301 total_capacity_.store(initial_capacity_, std::memory_order_release);
302 total_allocated_.store(0, std::memory_order_release);
303 allocation_count_.store(0, std::memory_order_release);
304 total_allocations_.store(0, std::memory_order_release);
305 total_deallocations_.store(0, std::memory_order_release);
306 alignment_waste_.store(0, std::memory_order_release);
307 block_count_.store(1, std::memory_order_release);
308}
